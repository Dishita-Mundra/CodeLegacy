// Real-time Load Balancing Logic
function balanceGrid(supply, demand) {
    if (demand > supply) {
        const deficit = demand - supply;
        console.log(`Alert: Drawing ${deficit}kW from Battery Storage`);
        activateRenewables(true);
    } else {
        const surplus = supply - demand;
        storeEnergy(surplus); // Recharge Grid Batteries
    }
}
