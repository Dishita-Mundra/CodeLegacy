// Smart Contract logic to prevent multiple votes
mapping(address => bool) public hasVoted;

function castVote(uint _candidateId) public {
    // 1. Check if voter already voted
    require(!hasVoted[msg.sender], "Voter has already voted.");
    
    // 2. Mark as voted
    hasVoted[msg.sender] = true;
    
    // 3. Update vote count securely
    candidates[_candidateId].voteCount++;
    
    emit votedEvent(_candidateId);
}

Voter Anonymity

The system decouples the Voter's Identity from the specific ballot cast using Zero-Knowledge Proofs.

Immutability

Once a vote is recorded on the blockchain, it cannot be deleted or modified by any authority.
